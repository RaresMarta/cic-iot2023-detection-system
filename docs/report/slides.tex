% ===== 2. Why IoT security matters ======================================
\begin{frame}{Why this matters}
  \vspace{0.6em}
  \begin{itemize}\itemsep1.1em
    \item Billions of IoT devices — cameras, plugs, sensors — shipped with
          weak, rarely-patched security.
    \item Each is an entry point for attackers: floods, scans, botnets like Mirai.
    \item The defence: detect attacks live, per flow.
  \end{itemize}
\end{frame}

% ===== 4. Dataset =======================================================
\begin{frame}{The data: CIC-IoT-2023}
  \centering
  {\large\textbf{Canadian Institute for Cybersecurity\\ University of New Brunswick \textnormal{(2023)}}}\par
  \vspace{1.2em}
  \begin{itemize}\itemsep1.0em
    \item 46.7M flows · 105 real IoT devices · 33 attacks in 7 families.
    \item 39 features — flow statistics, \emph{no payload}.
    \item Count features leak the capture window — I treat them as leakage.
  \end{itemize}
\end{frame}

% ===== 6a. Meet the families (define the 8 classes) =====================
\begin{frame}{What we classify — 8 families}
  \centering\renewcommand{\arraystretch}{1.45}
  {\small
  \begin{tabular}{ll}
    \textbf{Benign}     & the normal smart-home traffic we want to leave alone\\ \hline
    \textbf{DDoS}       & many devices flood a target at once\\
    \textbf{DoS}        & one machine floods a target until it falls over\\
    \textbf{Mirai}      & hacked IoT devices, taken over and remote-controlled\\
    \textbf{Recon}      & an attacker quietly mapping what's reachable\\
    \textbf{Spoofing}   & pretending to be someone else to redirect traffic\\
    \textbf{Web}        & attacks hidden inside web requests (SQL injection, XSS)\\
    \textbf{BruteForce} & guessing passwords over and over until one works\\
  \end{tabular}}
\end{frame}

  \vspace{1.2em}
  \begin{itemize}\footnotesize
    \item Class-weighted loss for imbalance; tree baseline (Random Forest) on the same split.
    \item One stratified random split — comparable to the literature, but optimistic.
  \end{itemize}

% ===== 12. Deployment choices ===========================================
\begin{frame}{Deployment choices — where each piece runs}
  \centering\renewcommand{\arraystretch}{1.3}
  \begin{tabular}{lll}
    \textbf{Piece} & \textbf{Runs on} & \textbf{Why}\\ \hline
    Detector + API   & FastAPI on DigitalOcean droplet & serves both MLP \& RF\\
    Live sensor      & same droplet (Docker) & captures real traffic\\
    Backplane        & Supabase (Postgres + Realtime) & push flows, no polling\\
    Dashboard        & React · Vite · Vercel & static, edge-served, CI deploy\\
  \end{tabular}
  \vspace{0.9em}
  \begin{itemize}\footnotesize
    \item Decoupled: training is offline; serving is a stateless API the UI calls over REST.
  \end{itemize}
\end{frame}

% ===== 14. End-to-end validation ========================================
\begin{frame}{End-to-end — known-label captures}
  Run real captures whose correct labels we know in advance, end to end:
  \begin{itemize}
    \item Benign browsing → \textbf{98\% Benign}
    \item SYN flood → \textbf{100\% Attack}, mean confidence 99.6\%, family = DoS
    \item Port scan → family = \textbf{Recon}
  \end{itemize}
  \vspace{0.6em}
  In-distribution the binary gate is reliable: \textbf{false-alarm rate 1.77\%}.
\end{frame}

% ===== 15. Conclusions ==================================================
\begin{frame}{Conclusions \& future work}
  \textbf{Delivered}
  \begin{itemize}\footnotesize
    \item A defensible MLP baseline with tree comparison on CIC-IoT-2023.
    \item A working live, streaming IDS: capture $\to$ classify $\to$ dashboard.
  \end{itemize}
  \textbf{Limitations}
  \begin{itemize}\footnotesize
    \item Out-of-distribution over-flagging (domain shift); dataset windowing leakage;
          transport-only features miss web attacks; optimistic random split.
  \end{itemize}
  \textbf{Future work}
  \begin{itemize}\footnotesize
    \item Leakage-aware retraining; fine-tune on representative traffic;
          source allow-listing / baselining; line-rate deployment.
  \end{itemize}
\end{frame}
